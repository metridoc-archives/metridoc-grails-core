115         * Indicates progress with the default progress indicator
116         */
117        public void indicateProgress() {
118            progressIndicatorActive = true;
119            if (isAnsiEnabled()) {
120                if (StringUtils.hasText(lastMessage)) {
121                    updateStatus(lastMessage + indicator);
122                }
123            } else {
124                logger.info(indicator);
125            }

131         * @param number The current number
132         * @param total  The total number
133         */
134        public void indicateProgress(int number, int total) {
135            progressIndicatorActive = true;
136            String currMsg = lastMessage;
137            try {
138                updateStatus(currMsg + ' ' + number + " of " + total);
139            } finally {

151            progressIndicatorActive = true;
152            String currMsg = lastMessage;
153            try {
154                int percentage = Math.round(NumberMath.multiply(NumberMath.divide(number, total), 100).floatValue());
155    
156                if (!isAnsiEnabled()) {
157                    logger.info("..");
158                    logger.info(String.valueOf(percentage) + '%');
159                } else {
160                    updateStatus(currMsg + ' ' + percentage + '%');
161                }
162            } finally {
163                lastMessage = currMsg;
164            }

277        public void error(String msg, Throwable error) {
278            try {
279                if ((isVerbose() || stacktrace) && error != null) {
280                    printStackTrace(msg, error);
281                    error(ERROR, msg);
282                } else {
283                    error(ERROR, msg + STACKTRACE_MESSAGE);
284                }
285            } finally {
286                postPrintMessage();
287            }
288        }

299        private void printStackTrace(String message, Throwable error) {
300    
301            if ((error instanceof BuildException) && error.getCause() != null) {
302                error = error.getCause();
303            }
304            if (!isVerbose() && !Boolean.getBoolean("grails.full.stacktrace")) {
305                StackTraceUtils.deepSanitize(error);
306            }
307            StringWriter sw = new StringWriter();
308            PrintWriter ps = new PrintWriter(sw);
309            message = message == null ? error.getMessage() : message;
310            if (!isVerbose()) {
311                message = message + STACKTRACE_FILTERED_MESSAGE;
312            }
313            ps.println(message);
314            error.printStackTrace(ps);
315            error(sw.toString());
316        }

361        /**
362         * Like {@link #userInput(String)} except that the user's entered characters will be replaced with ‘*’ on the CLI,
363         * masking the input (i.e. suitable for capturing passwords etc.).
364         *
365         * @param msg The message/question to display.
366         * @return The line of text entered by the user. May be a blank

369        public String secureUserInput(String msg) {
370            return doUserInput(msg, true);
371        }

373        private String doUserInput(String msg, boolean secure) {
374            // Add a space to the end of the message if there isn't one already.
375            if (!msg.endsWith(" ") && !msg.endsWith("\t")) {
376                msg += ' ';
377            }
378    
379            lastMessage = "";
380            return readLine(msg, secure);
381        }

402        /**
403         * Replacement for AntBuilder.input() to eliminate dependency of
404         * GrailsScriptRunner on the Ant libraries. Prints a message and
405         * list of valid responses, then returns whatever the user enters
406         * (once they press &lt;return&gt;). If the user enters something
407         * that is not in the array of valid responses, the message is
408         * displayed again and the method waits for more input. It will
409         * display the message a maximum of three times before it gives up
410         * and returns <code>null</code>.
411         *
412         * @param message        The message/question to display.
413         * @param validResponses An array of responses that the user is
414         *                       allowed to enter. Displayed after the message.
415         * @return The line of text entered by the user, or <code>null</code>
416         *         if the user never entered a valid string.
417         */
418        public String userInput(String message, String[] validResponses) {
419            if (validResponses == null) {
420                return userInput(message);
421            }
422    
423            String question = createQuestion(message, validResponses);
424            String response = userInput(question);
425            for (String validResponse : validResponses) {
426                if (response != null && response.equalsIgnoreCase(validResponse)) {
427                    return response;
428                }
429            }
430    
431            return userInput("Invalid input. Must be one of ", validResponses);
432        }
433    
434        private String createQuestion(String message, String[] validResponses) {
435            return message + "[" + DefaultGroovyMethods.join(validResponses, ",") + "] ";
436        }

438        public void error(String label, String message) {
439            if (message == null) {
440                return;
441            }
442    
443            try {
444                if (isAnsiEnabled()) {
445                    throw new RuntimeException("ANSI should not be enabled");
446                } else {
447                    logger.info(label);
448                    logger.info(" ");
449                    logSimpleError(message);
450                }
451            } finally {
452                postPrintMessage();
453            }
454        }
