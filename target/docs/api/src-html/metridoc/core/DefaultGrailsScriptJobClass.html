015    public class DefaultGrailsScriptJobClass extends DefaultGrailsJobClass {
016        public static final String RUN = "run";
017    
018        public DefaultGrailsScriptJobClass(Class clazz) {
019            super(clazz);
020        }
021    
022        @Override
023        public void execute() {
024            if(isScript()) {
025                executeScript();
026                return;
027            }
028    
029            super.execute();
030        }
031    
032        @Override
033        public void execute(JobExecutionContext context) {
034            if(isScript()) {
035                executeScript(context);
036                return;
037            }
038            super.execute(context);
039        }
040    
041        private boolean isScript() {
042            return Script.class.isAssignableFrom(getClazz());
043        }
044    
045        protected void executeScript() {
046            getMetaClass().invokeMethod(getReferenceInstance(), RUN, new Object[]{});
047        }
048    
049        protected void executeScript(JobExecutionContext context) {
050            Binding binding = (Binding) getMetaClass()
051                    .invokeMethod(getReferenceInstance(), "getBinding", new Object[]{});
052            binding.setVariable("jobExecutionContext", context);
053            getMetaClass().invokeMethod(getReferenceInstance(), RUN, new Object[]{});
054        }
055    }
