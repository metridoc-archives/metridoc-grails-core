032        /**
033         * creates a job based on the passed {@link TriggerFiredBundle}.
034         *
035         * @param bundle    the bundle to base the job off of
036         * @param scheduler the quartz scheduler
037         * @return the job
038         * @throws SchedulerException
039         */
040        @Override
041        public org.quartz.Job newJob(TriggerFiredBundle bundle, Scheduler scheduler) throws SchedulerException {
042            OperableTrigger trigger = bundle.getTrigger();
043            String triggerName = trigger.getKey().getName();
044            JobDataMap jobDataMap = bundle.getJobDetail().getJobDataMap();
045            JobDataMap triggerJobDataMap = trigger.getJobDataMap();
046            QuartzService quartzService = applicationContext.getBean("quartzService", QuartzService.class);
047            ConfigObject config = quartzService.getConfigByTriggerName(triggerName);
048            triggerJobDataMap.put("config", config);
049            String args = quartzService.getArgsByTriggerName(triggerName);
050            triggerJobDataMap.put("args", args);
051            String grailsJobName = (String) jobDataMap.get(GrailsArtefactJobDetailFactoryBean.JOB_NAME_PARAMETER);
052            org.quartz.Job job;
053    
054            if (grailsJobName != null) {
055                job = quartzService.buildJob(grailsJobName);
056            } else if (quartzService.isRemoteScriptJob(triggerName)) {
057                job = quartzService.buildRemoteScriptJob(triggerName);
058            } else {
059                job = super.newJob(bundle, scheduler);
060            }
061    
062            if (job instanceof GrailsArtefactJob) {
063                QuartzDisplayJob previousRun;
064                if (jobRuns.containsKey(triggerName)) {
065                    previousRun = jobRuns.get(triggerName);
066                } else {
067                    previousRun = new QuartzDisplayJob();
068                }
069    
070                QuartzDisplayJob displayJob;
071                try {
072                    displayJob = (QuartzDisplayJob) previousRun.clone();
073                } catch (CloneNotSupportedException e) {
074                    throw new RuntimeException(e);
075                }
076                displayJob.setQuartzService(quartzService);
077                displayJob.setSessionFactory(sessionFactory);
078                displayJob.setJob((GrailsArtefactJob) job);
079                displayJob.setJobKey(trigger.getJobKey());
080                displayJob.setTrigger(trigger);
081                job = displayJob;
082                jobRuns.put(triggerName, displayJob);
083            }
084            return job;
085        }

091        /**
092         * Quartz Job implementation that invokes execute() on the GrailsTaskClassJob instance whilst recording the time.
093         * All jobs are non concurrent
094         */
095        @DisallowConcurrentExecution
096        public static class QuartzDisplayJob implements org.quartz.Job, Cloneable {
097            static final Logger displayLogger = LoggerFactory.getLogger(QuartzDisplayJob.class);
098            private QuartzService quartzService;
099            private Long duration;
100            private boolean interrupting;
101            private JOB_STATUS status;
102            private Date lastRun;
103            private String error;
104            private String tooltip;
105            private Trigger.TriggerState triggerStatus;
106            private Trigger trigger;
107            private boolean manualJob = false;
108    
109            public enum JOB_STATUS {RUNNING, COMPLETE, ERROR, INTERRUPTING}
110    
111            GrailsArtefactJob job;
112            private SessionFactory sessionFactory;
113            private final Logger log = LoggerFactory.getLogger(QuartzDisplayJob.class);
114            private JobKey jobKey;
115    
116            public QuartzDisplayJob(GrailsArtefactJob job, SessionFactory sessionFactory) {
117                this.job = job;
118                this.sessionFactory = sessionFactory;
119            }
120    
121            public QuartzDisplayJob() {
122            }
123    
124            public void execute(final JobExecutionContext context) throws JobExecutionException {
125                try {
126                    Long lastDuration;
127                    String jobName = trigger.getKey().getName();
128                    lastDuration = QuartzService.getLastDuration(jobName);
129                    Date lastRunUsed = new Date();
130                    String lastTooltip = getTooltip();
131                    clearDetails();
132                    setDuration(lastDuration);
133                    setTooltip(lastTooltip);
134                    setLastRun(lastRunUsed);
135                    setStatus(JOB_STATUS.RUNNING);
136                    long start = System.currentTimeMillis();
137                    String error;
138                    try {
139                        displayLogger.info("starting job {} at {}", getTrigger().getKey().getName(), getLastRun());
140                        job.execute(context);
141                        sessionFactory.getCurrentSession().flush();
142                    } catch (Throwable e) {
143                        quartzService.mailJobError(e, context);
144                        error = e.getMessage();
145                        setError(error);
146                        setStatus(JOB_STATUS.ERROR);
147                        addDurationAndToolTip(start);
148                        displayLogger.error("Exception occurred running job " + getTrigger().getKey().getName(), e);
149    
150                        if (e instanceof JobExecutionException) {
151                            throw (JobExecutionException) e;
152                        }
153                        throw new JobExecutionException(e.getMessage(), e);
154                    }
155                    setStatus(JOB_STATUS.COMPLETE);
156                    addDurationAndToolTip(start);
157                } catch (Exception e) {
158                    //this is really annoying that we have to log this twice (aka quartz will log this too)!, but for the logging per job to work properly this must happen
159    
160                    if (e instanceof JobExecutionException) {
161                        throw (JobExecutionException) e;
162                    }
163                    throw new JobExecutionException(e.getMessage(), e);
164                } finally {
165                    org.quartz.Trigger currentTrigger = context.getTrigger();
166                    try {
167                        setInterrupting(false);
168                    } catch (UnableToInterruptJobException e) {
169                        String errorMessage = "Could not set interrupt to false for trigger " + currentTrigger.getKey().getName();
170                        displayLogger.warn(errorMessage);
171                    }
172                    JobDataMap jobData = currentTrigger.getJobDataMap();
173                    if (jobData.containsKey("oldTrigger")) {
174                        try {
175                            org.quartz.Trigger oldTrigger = (org.quartz.Trigger) jobData.get("oldTrigger");
176                            if (oldTrigger != null) {
177                                context.getScheduler().rescheduleJob(context.getTrigger().getKey(), oldTrigger);
178                            } else {
179                                context.getScheduler().unscheduleJob(currentTrigger.getKey());
180                            }
181                        } catch (SchedulerException e) {
182                            log.error("Could not reschedule trigger " + currentTrigger.getKey().getName(), e);
183                        }
184                    }
185                    try {
186                        quartzService.saveRun(this);
187                    } catch (Exception e) {
188                        displayLogger.error("Could not save run", e);
189                    }
190                }
191    
192            }
193    
194            public void addDurationAndToolTip(long start) {
195                long duration = System.currentTimeMillis() - start;
196                setDuration(duration);
197                addToolTip();
198            }
199    
200            public void addToolTip() {
201                if (duration != null) {
202                    String jobRunTime = "Most recent job ran in " + duration + "ms";
203                    String jobException = error != null ? ", with error " + error : "";
204                    String tooltip = jobRunTime + jobException;
205                    setTooltip(tooltip);
206                }
207            }
