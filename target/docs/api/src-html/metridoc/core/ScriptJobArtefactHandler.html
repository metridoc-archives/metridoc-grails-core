018    public class ScriptJobArtefactHandler extends ArtefactHandlerAdapter {
019    
020        JobArtefactHandler jobArtefactHandler = new JobArtefactHandler();
021    
022        public ScriptJobArtefactHandler() {
023            super(DefaultGrailsJobClass.JOB, GrailsJobClass.class, DefaultGrailsScriptJobClass.class, null);
024        }
025    
026        public boolean isArtefactClass(Class clazz) {
027            if(jobArtefactHandler.isArtefactClass(clazz)) return true;
028    
029            boolean assignableFrom = Script.class.isAssignableFrom(clazz);
030            boolean nameEndsWithJob = clazz.getName().endsWith(DefaultGrailsJobClass.JOB);
031            return assignableFrom && nameEndsWithJob;
032        }
033    }
