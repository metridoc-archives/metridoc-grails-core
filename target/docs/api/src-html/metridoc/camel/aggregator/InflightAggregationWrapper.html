016    package metridoc.camel.aggregator;
017    
018    import org.apache.camel.Exchange;
019    import org.apache.camel.processor.aggregate.AggregationStrategy;
020    import org.apache.camel.spi.Synchronization;
021    
022    /**
023     *
024     * @author tbarker
025     */
026    public class InflightAggregationWrapper implements AggregationStrategy {
027    
028        private Exchange currentExchange;
029        private AggregationStrategy wrappedStrategy;
030        private static final String METRIDOC_SYNCHRONIZOR = "MetridocSynchronizor";
031    
032    
033        public InflightAggregationWrapper(AggregationStrategy wrappedStrategy) {
034            this.wrappedStrategy = wrappedStrategy;
035        }
036    
037        public Exchange aggregate(Exchange oldExchange, Exchange newExchange) {
038            Exchange aggExchange = wrappedStrategy.aggregate(oldExchange, newExchange);
039    
040            if (aggExchange != null) {
041                if (aggExchange != currentExchange) {
042                    aggExchange.getContext().getInflightRepository().add(aggExchange);
043                    Synchronization synchronizor;
044                    if (currentExchange != null) {
045                        synchronizor = currentExchange.getProperty(METRIDOC_SYNCHRONIZOR, Synchronization.class);
046                        synchronizor.onComplete(currentExchange);
047                    }
048                    currentExchange = aggExchange;
049                    synchronizor = new AggregationSynchronizor();
050                    currentExchange.setProperty(METRIDOC_SYNCHRONIZOR, synchronizor);
051                    currentExchange.addOnCompletion(synchronizor);
052                }
053            }
054    
055            return aggExchange;
056        }
057    }
