020    package metridoc.camel.aggregator;
021    
022    import org.apache.camel.Exchange;
023    import org.apache.camel.impl.DefaultExchange;
024    import org.apache.camel.processor.aggregate.AggregationStrategy;
025    
026    import java.util.ArrayList;
027    import java.util.List;
028    
029    /**
030     *
031     * @author Thomas Barker
032     */
033    public class BodyAggregator implements AggregationStrategy {
034    
035        @Override
036        @SuppressWarnings("unchecked")
037        public Exchange aggregate(Exchange oldExchange, Exchange newExchange) {
038            
039            if (oldExchange == null) {
040                oldExchange = new DefaultExchange(newExchange);
041                oldExchange.getIn().setHeaders(newExchange.getIn().getHeaders());
042                List<Object> body = new ArrayList<Object>();
043                oldExchange.getIn().setBody(body);
044                oldExchange.getExchangeId();
045            }
046            oldExchange.getIn().getBody(List.class).add(newExchange.getIn().getBody());
047            
048            return oldExchange;
049        }
050    
051    }
